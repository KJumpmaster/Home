async function initiateSystem() { const overlay = document.getElementById('loading-overlay'); try { console.log("GEM: Running Clean Stabilizer (No Groups)..."); // 1. DATA ACQUISITION (Only Tree and Registry) const [treeData, regText] = await Promise.all([ fetch('./usa_tree_v4.json').then(r => { if (!r.ok) throw new Error('usa_tree_v4.json missing'); return r.json(); }), fetch('./SP_Registry_CANON_3_31.csv').then(r => { if (!r.ok) throw new Error('Registry CSV missing'); return r.text(); }) ]); // 2. PARSE REGISTRY const registry = {}; const lines = regText.split('\n'); for (let i = 1; i < lines.length; i++) { const cols = lines[i].split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/); if (cols.length > 2) { const id = cols[0]?.replace(/"/g, '').toLowerCase().trim(); const img = cols[2]?.replace(/"/g, '').trim(); if (id) registry[id] = img; } } // 3. RENDER GRID const theater = document.getElementById('research-theater'); theater.innerHTML = ''; treeData.cells.forEach(cell => { const id = cell.cell_id.toLowerCase(); const imageName = registry[id] || 'placeholder.png'; const card = document.createElement('div'); card.className = 'v-card'; // Regular card only card.style.gridColumn = cell.column; card.style.gridRow = cell.row; card.innerHTML = `
${(cell.display_name || cell.cell_id).replace(/_/g, ' ')}
`; theater.appendChild(card); }); console.log("GEM: System Stable. Grid Rendered."); overlay.style.display = 'none'; } catch (err) { console.error("STABILIZER ERROR:", err); overlay.innerHTML = `

OFFLINE

${err.message.toUpperCase()}

`; } }
