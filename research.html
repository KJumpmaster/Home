function renderCells(cells, rowCountsByRank, rankLayout) { const rectMap = new Map(); for (const cell of cells) { const rect = getCellRect(cell, rowCountsByRank, rankLayout); rectMap.set(cell.cell_id, rect); const wrap = document.createElement("div"); wrap.className = "cell-wrap" + (cell.type === "group" ? " grouped" : ""); wrap.style.left = rect.x + "px"; wrap.style.top = rect.y + "px"; if (cell.type === "group") { const s2 = document.createElement("div"); s2.className = "cell-shadow-2"; const s1 = document.createElement("div"); s1.className = "cell-shadow-1"; wrap.appendChild(s2); wrap.appendChild(s1); } wrap.appendChild(createCellElement(cell)); nodeLayerEl.appendChild(wrap); } return rectMap; }
