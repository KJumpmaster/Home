// THE MASTER DATA LINK const PIC_BASE = "https://raw.githubusercontent.com/JumpingKJ/KJ_Project_NORDEN/main/Graphics/Planes/Icons/"; async function initiateSystem() { const overlay = document.getElementById('loading-overlay'); try { console.log("GEM: Initiating Handshake..."); // 1. ACQUIRE ASSETS (No shop.blkx - Lightweight Only) const [treeData, groupData, regText] = await Promise.all([ fetch('./usa_tree_v4.json').then(r => r.json()), fetch('./usa_groups.json').then(r => r.json()), fetch('./SP_Registry_CANON_3_31.csv').then(r => r.text()) ]); // 2. MAP REGISTRY (ID to Image Filename) const registry = {}; const lines = regText.split('\n'); for (let i = 1; i < lines.length; i++) { const cols = lines[i].split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/); if (cols.length > 2) { const id = cols[0].replace(/"/g, '').trim().toLowerCase(); const img = cols[2].replace(/"/g, '').trim(); registry[id] = img; } } // 3. RENDER THEATER const theater = document.getElementById('research-theater'); theater.innerHTML = ''; treeData.cells.forEach(cell => { if (!cell.cell_id) return; const id = cell.cell_id.toLowerCase(); // Check if this cell is a Group/Folder const squadMembers = groupData[id] || null; const isGroup = squadMembers !== null; // Image Selection Logic let imageName = registry[id] || 'placeholder.png'; if (isGroup && !imageName.includes('Group')) { // If it's a group, try to force the Group-style icon imageName = imageName.split('.')[0] + ' Group.png'; } // Create the Card Element const card = document.createElement('div'); card.className = `v-card ${isGroup ? 'group-stack' : ''}`; card.style.gridColumn = cell.column; card.style.gridRow = cell.row; // Generate Squadron List if applicable let manifestHtml = isGroup ? `
${squadMembers.join('
')}
` : ""; card.innerHTML = `
${(cell.display_name || cell.cell_id).replace(/_/g, ' ')}
${manifestHtml} `; theater.appendChild(card); }); // 4. SYSTEM ONLINE overlay.style.display = 'none'; console.log("GEM: 34 Groups Successfully Deployed."); } catch (err) { console.error("DATA LINK FAILURE:", err); overlay.innerHTML = `

SYSTEM OFFLINE

${err.message.toUpperCase()}

`; } } // Start the sequence window.onload = initiateSystem;
