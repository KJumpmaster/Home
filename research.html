// THE MASTER DATA LINKS const PIC_BASE = "https://raw.githubusercontent.com/JumpingKJ/KJ_Project_NORDEN/main/Graphics/Planes/Icons/"; async function initiateSystem() { const overlay = document.getElementById('loading-overlay'); try { console.log("GEM: Initiating 100% Combat-Ready Handshake..."); // 1. DATA ACQUISITION (The Triple Threat) const [treeRes, groupRes, regRes] = await Promise.all([ fetch('./usa_tree_v4.json'), fetch('./usa_groups.json'), fetch('./SP_Registry_CANON_3_31.csv') ]); // Error checking to make sure your files are actually there if (!treeRes.ok) throw new Error("usa_tree_v4.json MISSING"); if (!groupRes.ok) throw new Error("usa_groups.json MISSING"); if (!regRes.ok) throw new Error("SP_Registry_CANON_3_31.csv MISSING"); const treeData = await treeRes.json(); const groupData = await groupRes.json(); const regText = await regRes.text(); // 2. MAP REGISTRY (ID to Image Filename) const registry = {}; const lines = regText.split('\n'); for (let i = 1; i < lines.length; i++) { const cols = lines[i].split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/); if (cols.length > 2) { const id = cols[0].replace(/"/g, '').trim().toLowerCase(); const img = cols[2].replace(/"/g, '').trim(); if (id) registry[id] = img; } } // 3. RENDER THEATER const theater = document.getElementById('research-theater'); theater.innerHTML = ''; treeData.cells.forEach(cell => { if (!cell.cell_id) return; const id = cell.cell_id.toLowerCase(); // GROUP LOGIC: Does this aircraft have "wingmen" in the groups file? const squadMembers = groupData[id] || null; const isGroup = squadMembers !== null; // IMAGE SELECTION: If it's a group, force the "Group.png" icon let imageName = registry[id] || 'placeholder.png'; if (isGroup && !imageName.includes('Group')) { imageName = imageName.split('.')[0] + ' Group.png'; } // CREATE THE CARD const card = document.createElement('div'); // 'group-stack' adds the double-border CSS you built card.className = `v-card ${isGroup ? 'group-stack' : ''}`; card.style.gridColumn = cell.column; card.style.gridRow = cell.row; // MANIFEST LIST: The list of planes that shows up inside the folder let manifestHtml = ""; if (isGroup) { manifestHtml = `
${squadMembers.join('
')}
`; } // FINAL INJECTION card.innerHTML = `
${(cell.display_name || cell.cell_id).replace(/_/g, ' ')}
${manifestHtml} `; theater.appendChild(card); }); // 4. MISSION SUCCESS overlay.style.display = 'none'; console.log("GEM: SYSTEM 100% OPERATIONAL. HOORAY!"); } catch (err) { console.error("CRITICAL FAILURE:", err); overlay.innerHTML = `

OFFLINE

${err.message.toUpperCase()}

Verify your filenames in the folder!

`; } } // Kickstart the system window.onload = initiateSystem;
