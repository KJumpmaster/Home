path.setAttribute( "d", `M ${x1} ${y1} L ${x1} ${midY} L ${x2} ${midY} L ${x2} ${y2}` ); path.setAttribute("class", "tree-edge"); edgeLayer.appendChild(path); }); } window.onload = initiateSystem; window.addEventListener('resize', () => { const theater = document.getElementById('research-theater');
