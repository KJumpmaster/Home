// THE MASTER DATA LINK const PIC_BASE = "https://raw.githubusercontent.com/JumpingKJ/KJ_Project_NORDEN/main/Graphics/Planes/Icons/"; async function initiateSystem() { const overlay = document.getElementById('loading-overlay'); try { console.log("GEM: Initiating Ironclad Stable Build..."); // 1. ACQUIRE ASSETS (Tree and Registry Only) // Strictly using only these two files to guarantee no crashes. const [treeRes, regRes] = await Promise.all([ fetch('./usa_tree_v4.json'), fetch('./SP_Registry_CANON_3_31.csv') ]); if (!treeRes.ok) throw new Error("usa_tree_v4.json is missing or renamed"); if (!regRes.ok) throw new Error("SP_Registry_CANON_3_31.csv is missing or renamed"); const treeData = await treeRes.json(); const regText = await regRes.text(); // 2. MAP REGISTRY (ID to Image Filename) const registry = {}; const lines = regText.split('\n'); for (let i = 1; i < lines.length; i++) { const cols = lines[i].split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/); if (cols.length > 2) { const id = cols[0].replace(/"/g, '').trim().toLowerCase(); const img = cols[2].replace(/"/g, '').trim(); if (id) registry[id] = img; } } // 3. RENDER THEATER const theater = document.getElementById('research-theater'); theater.innerHTML = ''; treeData.cells.forEach(cell => { if (!cell.cell_id) return; const id = cell.cell_id.toLowerCase(); const imageName = registry[id] || 'placeholder.png'; const card = document.createElement('div'); card.className = 'v-card'; // Single-card mode only card.style.gridColumn = cell.column; card.style.gridRow = cell.row; // This HTML structure is the one that gave us the 100% success card.innerHTML = `
${(cell.display_name || cell.cell_id).replace(/_/g, ' ')}
`; theater.appendChild(card); }); // 4. SYSTEM ONLINE overlay.style.display = 'none'; console.log("GEM: System 100% Stable. Grid Rendered."); } catch (err) { console.error("DATA LINK FAILURE:", err); overlay.innerHTML = `

OFFLINE

${err.message.toUpperCase()}

`; } } // Start the sequence immediately window.onload = initiateSystem;
